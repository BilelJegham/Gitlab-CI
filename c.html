Utilisation de Makefile pour compiler et lancer un projet en C.

Makefile

run: compile
    ./run
compile: 
    gcc **.c -ldl -o run
test:
    cppcheck .
install:
    apt-get -qq update && apt-get -qq install -y cppcheck

.gitlab-ci.yml

image: gcc

stages:
  - build
  - test
  - run

compile:
  stage: build
  script:
    - make compile

run:
  stage: run
  script:
   - make run

test: 
  stage: test
  before_script:
    - make install
  script:
    - make test
